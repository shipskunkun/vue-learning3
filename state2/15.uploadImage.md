```js
const express = require('express');
const multer = require('multer');
const cors = require('cors');
const upload = multer({ dest: 'uploads/' });

const app = express();

app.get('/', (req, res) => {
  res.send('hello nodejs');
});

app.options('/upload', cors());
app.post('/upload', cors(), upload.single('file'), (req, res) => {
  res.send(req.file.filename);
});

app.get('/preview/:key', cors(), (req, res) => {
  res.sendFile(
    `uploads/${req.params.key}`,
    {
      root: __dirname,
      headers: {
        'Content-Type': 'image/jpeg',
      },
    },
    error => {
      console.log(error);
    }
  );
});

var port = process.env.PORT || 3000;
console.log(port);
app.listen(port);
```

问题：如何区分两个完全一样的文件

方法二：前端上传后名字，每次累加1， a.jpg,   a(1).jpg   

a(2).jpg //比较麻烦

a(1)(1).jpg  // 确保名字唯一性即可

```js
generateName (name) {
  while (this.fileList.filter(f => f.name === name).length > 0) {
  let dotIndex = name.lastIndexOf('.')
  let nameWithoutExtension = name.substring(0, dotIndex)
  let extension = name.substring(dotIndex)
  name = nameWithoutExtension + '(1)' + extension
  }
  return name
}
```

#### 15-12视频: 渲染任务是异步执行的！

bug原因，更新了 fileList 数据，但是页面并没有发送变化

子组件更新时机？

在子组件中，通过 emit 修改父元素的 fileList，这个过程是同步的

但是父组件中拿到 fileList 没有重新传给子组件

并没有重新render 子组件，所以子组件中的 fileList 不是最新的

重新render 是异步的

vue 的事件是同步的，ui更新是异步的，放到事件队列中



解决方法：

fail 放在异步函数中
