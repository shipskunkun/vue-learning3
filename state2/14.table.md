var d = value.list||[];

for(var i=0;i<d.length;i++){
  var add = true;
  if(arr.length>0){
    for(var j=0;j<arr.length;j++){
      if(arr[j][0].repeat_mark == d[i].repeat_mark){
        arr[j].push(d[i]);
        add = false;
